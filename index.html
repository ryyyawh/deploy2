Debounce Function

Prevent a function from being called too frequently in rapid events like keystrokes.

function debounce(func, delay) {
  let timer;
  return (...args) => {
    clearTimeout(timer);
    timer = setTimeout(() => func.apply(this, args), delay);
  };
}
